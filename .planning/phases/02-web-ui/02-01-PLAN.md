    Modify the viewBook function in internal/controller/http/web/books.go:

    Add highlight fetch after the stats fetch (follow the exact same pattern):

    ```go
    func (r *booksRoutes) viewBook(c *gin.Context) {
        bookID := c.Param("bookID")

        book, err := r.shelf.ViewBook(c.Request.Context(), bookID)
        if err != nil {
            c.HTML(500, "error", passStandartContext(c, gin.H{"error": err.Error()}))
            return
        }

        bookStats, err := r.stats.GetBookStats(c.Request.Context(), book.DocumentID)
        if err != nil {
            r.logger.Error(err, "failed to get book stats")
            bookStats = &stats.BookStats{} // Use empty stats in case of error
        }

        // NEW: Fetch highlights using the same error handling pattern
        highlights, err := r.highlight.Fetch(c.Request.Context(), book.DocumentID)
        if err != nil {
            r.logger.Error(err, "failed to fetch highlights")
            highlights = []entity.Highlight{} // Use empty slice on error, NOT nil
        }

        c.HTML(200, "book", passStandartContext(c, gin.H{
            "book":       book,
            "stats":      bookStats,
            "highlights": highlights,  // NEW
        }))
    }
    ```

    CRITICAL: Initialize to `[]entity.Highlight{}` on error, not nil. Nil causes template rendering issues.
    Use `book.DocumentID` (the MD5 hash), NOT `book.ID` (the UUID).
